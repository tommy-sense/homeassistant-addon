#include <tunables/global>

profile tommysense-homeassistant-addon flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability net_bind_service,
  capability setgid,
  capability setuid,
  file,
  signal (send) set=(kill,term,int,hup,cont),
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # Ptrace permissions for process monitoring
  ptrace (read) peer=tommysense-homeassistant-addon//caddy_service,
  ptrace (read) peer=tommysense-homeassistant-addon//tommy,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Data directory
  /data/** rw,

  # Start profiles for services
  /usr/bin/caddy cx -> caddy_service,
  /app/start.sh cx -> tommy,
  /usr/bin/node cx -> tommy,

  profile caddy_service flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_tommysense-homeassistant-addon,

    # Allow parent process to ptrace this profile
    ptrace (read, trace) peer=*_tommysense-homeassistant-addon,

    # Caddy binary
    /usr/bin/caddy rix,
    
    # Caddy config
    /tmp/Caddyfile r,
    /tmp/** r,
    
    # Caddy data and cache directories
    /root/.config/ rwk,
    /root/.config/caddy/ rwk,
    /root/.config/caddy/** rwk,
    /root/.local/ rwk,
    /root/.local/share/ rwk,
    /root/.local/share/caddy/ rwk,
    /root/.local/share/caddy/** rwk,
    
    # Network
    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
    capability net_bind_service,
    capability net_admin,
    
    # System files
    /etc/mime.types r,
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,
    /etc/ssl/** r,
    /etc/ssl/certs/ r,
    /etc/ssl/certs/** r,
    
    # CA certificates
    /usr/share/ca-certificates/** r,
    
    # Proc and sys access
    /proc/*/mountinfo r,
    /proc/*/cgroup r,
    /proc/*/cpuset r,
    /proc/sys/net/** r,
    /proc/** r,
    /sys/fs/cgroup/cpu.max r,
    /sys/fs/cgroup/memory.max r,
    /sys/** r,
    
    # Device access
    /dev/tty rw,
    /dev/null rw,
    /dev/urandom r,
    /dev/random r,
    
    # Unix sockets
    unix,
  }

  profile tommy flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_tommysense-homeassistant-addon,

    # Allow parent process to ptrace this profile
    ptrace (read, trace) peer=*_tommysense-homeassistant-addon,

    # Data directory
    /data/** rw,
  
    # Application files
    /app/** rwix,

    # System binaries
    /bin/** rix,
    /usr/bin/** rix,
    /usr/lib/** rix,
    /usr/local/** rix,
    
    # Shell access
    /bin/sh rix,
    /bin/bash rix,
    /bin/dash rix,
    /usr/bin/sh rix,
    /usr/bin/bash rix,
    /usr/bin/dash rix,
    /usr/bin/getconf rix,
    /bin/getconf rix,
    
    # Network access
    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
    
    # Process control
    signal (send) set=(kill,term,int,hup,cont),
    capability setgid,
    capability setuid,
    capability net_bind_service,
    
    # System files
    /etc/passwd r,
    /etc/group r,
    /etc/hosts r,
    /etc/nsswitch.conf r,
    /etc/resolv.conf r,
    /etc/ssl/openssl.cnf r,
    /etc/ssl/** r,
    /etc/ld.so.cache r,
    /dev/tty rw,
    /dev/null rw,
    
    # Proc access - cgroup and stat
    /proc/sys/net/ipv4/ip_local_port_range r,
    /proc/uptime r,
    /proc/stat r,
    /proc/*/stat r,
    /proc/*/cgroup r,
    /proc/meminfo r,
    /proc/loadavg r,
    /proc/cpuinfo r,
    /proc/** r,
    
    # Sys access - cgroup and CPU frequency
    /sys/devices/system/cpu/online r,
    /sys/devices/system/cpu/cpufreq/*/scaling_cur_freq r,
    /sys/fs/cgroup/memory.max r,
    /sys/fs/cgroup/memory.high r,
    /sys/fs/cgroup/cpu.max r,
    /sys/** r,
  }
}
